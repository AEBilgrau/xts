Plotting Basics

Load the xts package and some sample data

library(xts)
## Loading required package: zoo
## 
## Attaching package: 'zoo'
## 
## The following objects are masked from 'package:base':
## 
##     as.Date, as.Date.numeric
data(sample_matrix)
sample.xts <- as.xts(sample_matrix)

Create a basic plot

plot(sample.xts[,4])
